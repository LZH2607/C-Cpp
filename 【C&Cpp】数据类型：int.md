# 【C&Cpp】数据类型：int

int：32位

范围：-2147483648 ~ 2147483647

示例程序：

```c++
#include <iostream>

using namespace std;

int main() {
	int a[5] = { -2147483648 , -1, 0, 1, 2147483647 };
	for (int i = 0; i < 5; i++) {
		for (int j = 0; j < 5; j++) {
			int b = a[i] * a[j];
			cout << b << "\t\t";
			//cout << hex << b << "\t\t";
		}
		cout << endl;
	}

	return 0;
}
```
